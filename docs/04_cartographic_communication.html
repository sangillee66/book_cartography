4.3.1.6 질감 texture

library(ggpattern)
library(gridpattern)
library(magick)

# 1) 구 이름을 map_id처럼 사용할 범주 변수로 확보
seoul_gu2 <- seoul_gu |>
  mutate(gu = as.character(SGG1_FNM))

# 2) gu 레벨 수만큼 magick 패턴을 뽑아(무작위) 매핑 벡터를 만든다
set.seed(20260227)
lev <- sort(unique(seoul_gu2$gu))

# gridpattern::names_magick 에서 구 개수만큼 "중복 없이" 뽑기
# (names_magick 길이가 구 개수보다 짧으면 replace=TRUE로 바꾸면 됩니다)
picked <- sample(gridpattern::names_magick, size = length(lev), replace = FALSE)

# named vector: (이름=gu) -> (값=pattern name)
pat_map <- setNames(picked, lev)

# 3) sf + geom_sf_pattern
my_graph <- ggplot(seoul_gu2) +
  geom_sf_pattern(
    aes(pattern_type = gu),
    pattern              = "magick",
    pattern_fill         = "black",
    pattern_aspect_ratio = 1.75,
    fill                 = "white",
    colour               = "black"
  ) +
  coord_sf() +
  theme_bw() +
  # 핵심: choices에 “구이름 -> magick패턴” 매핑을 넣는다
  scale_pattern_type_discrete(choices = pat_map) +
  theme(legend.position = "none")
